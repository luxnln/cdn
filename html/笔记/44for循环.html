数组去重
    var aList = [1,2,3,4,3,2,1,2,3,4,5,6,5,3,2,2,2,3,2,1,2,4,5,7];
    var aList2 = [];

    for(var i = 0; i < aList.length; i++){
       if(aList.indexOf(aList[i]) == i){
          aList2.push(aList[i]);
       }
    }

    alert(aList2);
