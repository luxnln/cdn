操作元素属性

获取的页面元素，就可以对页面元素的属性进行操作，属性的操作包括属性的读和写。
操作属性的方法
  1. "."操作
  2. "[]"操作
属性写法
  1. html的属性和js里面属性写法一样
  2. "class"属性成"className"
  3. "style"属性里面的属性，有横杆的改成驼峰式，比如："font_size"，改成"style.fontSize
通过"."操作属性：
    <script type="text/javascript">
       windo.onload = function(){
          var oDiv1 = document.getElementById('div');
          var oA = document.getElementById('link1');
          var oDiv2 = document.getElementById('box');
          //读取属性值
          var sId = oDiv1.id;
          alert(sId);
          //写属性
          oDiv.style.color = 'red';
          oA.href = "http://www.baidu.com";
          oA.title = "这是去到百度网的链接";
          oA.target = "_blank";
      
          //操作class属性需要写成className
          oDiv2.className = "box2";
       }
    </script>
    ...
    <style type="text/css">
       .box{
          font-size:20px;
          color:gold;
       }
       .box2{
          font-size:30px;
          color:pink;
       }
    </style>
    ...
    <div id ="div">这是一个div元素</div>
    <a href="#" id="link">这是一个链接</a>
    <div id ="box">这是一个div元素</div>
    ...
通过"[]"操作属性：
    <script type="text/javascript">
       windo.onload = function(){
          var oDiv = document.getElementById('div');
          var sMystyle = 'color';
          var sValue = 'red';
          //可以是：oDiv.style.color = sValue;
          //属性用变量来代替的话不可以是：oDiv.style.sMystyle = sValue;需要用[]来操作
          oDiv.style[sMystyle] = sValue;
          }
    </script>
    ...
    <div id ="div">这是一个div元素</div>
    ...
innerHTML
innerHTML可以读取或者写入标签包裹的内容
    <script type="text/javascript">
       windo.onload = function(){
          var oDiv = document.getElementById('div');
          //读取
          var sTxt = oDiv.innerHTML;
          alert(sTxt);
          //写入,原来的内容会被替换
          oDiv.innerHTML = "<href='http://www.baidu.com'>百度</a>";
          }
    </script>
    ...
    <div id ="div">这是一个div元素</div>
